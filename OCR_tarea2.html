import cv2
import pytesseract

pytesseract.pytesseract.tesseract_cmd = r"C:\Users\ADMIN\Downloads\Data\tesseract.exe"

"lista de placas aceptadas en el estacionamiento del CBtis144"
placas_autorizadas = ["PCH-96-04", "WLU-94-69","A00-AAA","ADH-123", "79-SX7", "UOR420"]

# cargar 2 imagenes de placas
imagenes = ["placa3.jpg", "placareal.jpg"]

placas_leidas = []   # lista donde guardaremos las placas detectadas

for imagen in imagenes:
    # cargar placa a leer
    placa = cv2.imread(imagen)
    placanueva = cv2.resize(placa, (900,270))

    # convertir a escala de grises
    placa_gris = cv2.cvtColor(placanueva, cv2.COLOR_BGR2GRAY)

    # separar el fondo del texto
    placa_res = cv2.threshold(placa_gris, 120, 255, cv2.THRESH_BINARY)[1]

    # configurar OCR
    placa_texto = pytesseract.image_to_string(placa_res, lang="spa")

    # limpiar la cadena de texto
    placa_texto_limpio = placa_texto.strip().replace(" ", "")

    # guardar la placa le칤da
    placas_leidas.append(placa_texto_limpio)

    # buscar la placa en la lista autorizada
    if placa_texto_limpio in placas_autorizadas:
        print("ACCESO PERMITIDO!! \nBIENVENIDO AL ESTACIONAMIENTO DEL CBTIS 144")
    else:
        print("PLACA NO REGISTRADA")

    # imprimir la placa le칤da
    print("Placa le칤da:", placa_texto_limpio)
    print("----------------------------------")

    # mostrar la imagen analizada
    cv2.imshow("imagen analizada", placanueva)
    cv2.waitKey(1000)

    cv2.destroyAllWindows()

# mostrar las placas le칤das
print("\nPlacas le칤das:", placas_leidas)

# mostrar las placas restantes (las que no se escanearon)
placas_restantes = []
for p in placas_autorizadas:
    if p not in placas_leidas:
        placas_restantes.append(p)

print("Placas no escaneadas:", placas_restantes)
